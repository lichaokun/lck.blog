# 1. 使用node截图

// 主要依赖puppeteer来操作
const puppeteer = require('puppeteer');
const path = require('path');
const screenshot = path.resolve(__dirname, '../screenshot');

(async () => {
    // 开启一个服务
    const browser = await puppeteer.launch();
    const page = await browser.newPage();

    // 去百度页面截图
    await page.goto('https://www.baidu.com');

    // 存到对应的目录下
    await page.screenshot({
        path: `${screenshot}/${Date.now()}.png`
    });
    
    // 关闭服务
    await browser.close();
})();

# 2. 爬下百度图片

// crawler.js
const puppeteer = require('puppeteer');
const path = require('path');
// 爬下来的图片存到对应的images文件夹
const images = path.resolve(__dirname, '../images');
// 根据图片的类型来使用对应的方法转化
const srcToImg = require('./utils');

(async () => {
    // 开启服务，去百度图片
    const browser = await puppeteer.launch();
    const page = await browser.newPage();
    await page.goto('https://image.baidu.com/');

    // 设置浏览器的宽高，模拟用户滚动来实现懒加载
    await page.setViewport({
        width: 1920,
        height: 1080
    });

    // 找到对应的输入框，输入人，然后点击
    await page.focus('#kw');
    await page.keyboard.sendCharacter('人');
    await page.click('.s_search');

    // 等页面加载完毕后
    page.on('load', async () => {
        // 使用evaluate 方法来和jq一样操作dom
        const srcs = await page.evaluate(() => {
            const imagesDom = document.querySelectorAll('img.main_img');
            // 转化类数组
            return Array.prototype.map.call(imagesDom, img => img.src);
        });

        // 存入images目录
        for (let i = 0; i < srcs.length; i++) {
            // sleep
            await page.waitFor(200);
            await srcToImg(srcs[i], images);
        }

        // 关闭浏览器
        await browser.close();
    });
})();

// utils.js
const fs = require('fs');
const http = require('http');
const https = require('https');
const path = require('path');
const { promisify } = require('util');
const writeFile = promisify(fs.writeFile);

// 根据类型来判断转化
module.exports = async (src, dir) => {
    if (/\.(jpg|png|gif)$/.test(src)) {
        await urlToImg(src, dir);
    } else {
        await base64ToImg(src, dir);
    }
};

// url => image
const urlToImg = promisify((url, dir, callback) => {
    const mod = /^https:/.test(url) ? https : http;
    const ext = path.extname(url);
    const file = path.join(dir, `${Date.now()}${ext}`);

    mod.get(url, res => {
        res.pipe(fs.createWriteStream(file))
            .on('finish', () => {
                callback();
                console.log(file);
            })
    });
});

// base64 => image
const base64ToImg = async function (base64Str, dir) {
    // data:image/jpeg;base64,/asdasda
    const matches = base64Str.match(/^data:(.+?);base64,(.+)$/);
    try {
        const ext = matches[1].split('/')[1].replace('jpeg', 'jpg');
        const file = path.join(dir, `${Date.now()}.${ext}`);
        await writeFile(file, matches[2], 'base64');
        console.log(file);
    } catch (ex) {
        console.log('非法 base64 字符串');
    }
};
