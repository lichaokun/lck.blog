// 迭代器
const makeIterator = (arr) => {
    let nextIndex = 0
    return {
        next: () => {
            if(nextIndex < arr.length) {
                return {
                    value: arr[nextIndex++],
                    done: false
                }
            }else {
                return {
                    done: true
                }
            }
        }
    }
}

const it = makeIterator(['吃饭', '睡觉', '打豆豆'])
console.log("1", it.next().value) // 吃饭
console.log("2", it.next().value) // 睡觉
console.log("3", it.next().value) // 打豆豆
console.log("4", it.next().done)  // true

// 生成器, 简化迭代器
function * makeIterator (arr) {
    for(let i = 0; i < arr.length; i++) {
        // 使用yield关键字实现函数暂停,并且会返回当前函数的状态
        yield arr[i]
    }
}
const gen = makeIterator(['吃饭', '睡觉', '打豆豆'])
console.log("1", gen.next().value) // 吃饭
console.log("2", gen.next().value) // 睡觉

# 2. Koa源码解析

Koa官方文档中文指南

  • application.js 是整个koa2 的入口文件，封装了context，request，response，以及最核心的中间件处理流程。
  • context.js 处理应用上下文，里面直接封装部分request.js和response.js的方法
  • request.js 处理http请求
  • response.js 处理http响应

// 2.6.2版本的Application.js 源码  删除了注释,依赖等一些方法
module.exports = class Application extends Emitter {
    constructor() {
        super();

        this.proxy = false;
        this.middleware = [];
        this.subdomainOffset = 2;
        this.env = process.env.NODE_ENV || 'development';
        this.context = Object.create(context);
        this.request = Object.create(request);
        this.response = Object.create(response);
        if (util.inspect.custom) {
            this[util.inspect.custom] = this.inspect;
        }
    }

    listen(...args) {
        // 通过node.js的http.createServer生成了一个服务器实例 
        const server = http.createServer(this.callback());
        // 监听端口号
        return server.listen(...args);
    }

    use(fn) {
        // push到中间件数组
        this.middleware.push(fn);
        return this;
    }

    callback() {
        // 通过compose处理中间件数组
        const fn = compose(this.middleware);

        if (!this.listenerCount('error')) this.on('error', this.onerror);

        const handleRequest = (req, res) => {
            // 生成上下文
            const ctx = this.createContext(req, res);
            return this.handleRequest(ctx, fn);
        };
        // 传入到this.listen方法中的callback
        return handleRequest;
    }

    handleRequest(ctx, fnMiddleware) {
        const res = ctx.res;
        res.statusCode = 404;
        const onerror = err => ctx.onerror(err);
        // 处理响应，通过respond方法返回body
        const handleResponse = () => respond(ctx);
        onFinished(res, onerror);
        // 逐层处理中间件然后交给handleResponse
        return fnMiddleware(ctx).then(handleResponse).catch(onerror);
    }

    createContext(req, res) {
        // 赋值
        const context = Object.create(this.context);
        const request = context.request = Object.create(this.request);
        const response = context.response = Object.create(this.response);
        context.app = request.app = response.app = this;
        context.req = request.req = response.req = req;
        context.res = request.res = response.res = res;
        request.ctx = response.ctx = context;
        request.response = response;
        response.request = request;
        context.originalUrl = request.originalUrl = req.url;
        context.state = {};
        return context;
    }
};

function respond(ctx) {
    // allow bypassing koa
    if (false === ctx.respond) return;

    if (!ctx.writable) return;

    const res = ctx.res;
    let body = ctx.body;
    const code = ctx.status;

    // ignore body
    if (statuses.empty[code]) {
        // strip headers
        ctx.body = null;
        return res.end();
    }

    if ('HEAD' == ctx.method) {
        if (!res.headersSent && isJSON(body)) {
            ctx.length = Buffer.byteLength(JSON.stringify(body));
        }
        return res.end();
    }

    // status body
    if (null == body) {
        if (ctx.req.httpVersionMajor >= 2) {
            body = String(code);
        } else {
            body = ctx.message || String(code);
        }
        if (!res.headersSent) {
            ctx.type = 'text';
            ctx.length = Buffer.byteLength(body);
        }
        return res.end(body);
    }

    // responses
    if (Buffer.isBuffer(body)) return res.end(body);
    if ('string' == typeof body) return res.end(body);
    if (body instanceof Stream) return body.pipe(res);

    // body: json
    body = JSON.stringify(body);
    if (!res.headersSent) {
        ctx.length = Buffer.byteLength(body);
    }
    res.end(body);
}

总结

  • 在koa里面一切的流程都是中间件
  • 1个http请求进入到koa中后，都会流经预先配置好的中间件
  • 在中间件执行的策略中，会先通过koacomponse来把这些中间件组合在一起， 一个接一个的把数组里面的函数依次执行，通过next的回调函数，不断的把控制权往下传递
  • 每个中间件都会拿到http请求的上下文(ctx)，通过ctx可以访问req,res上面的属性和方法,ctx,req,res互相引用

# 4-2 POST请求

const Koa = require('koa')
const app = new Koa()

// 获取Post请求的步骤：
// 解析上下文ctx中的原生nodex.js对象req。
// 将POST表单数据解析成query string-字符串.(例如:user=jspang&age=18)
// 将字符串转换成JSON格式。

// ctx.request和ctx.req的区别
// request是Koa2中context经过封装的请求对象,req是context提供的node.js原生HTTP请求对象

// 通过处理原生req来处理参数
const parseData = ctx => {
    // 把字符串转化为json对象
    const parseQueryStr = queryStr => {
        let queryData = {};
        let queryStrList = queryStr.split('&');
        for (let [index,item] of queryStrList.entries()) {
            let obj = item.split('=');
            queryData[obj[0]] = decodeURIComponent(obj[1]);
        }
        return queryData
    }
    return new Promise((resolve, reject) => {
        try {
            let postData = '';
            ctx.req.on('data', (data) => {
                postData += data
            })
            ctx.req.addListener("end", () => {
                resolve(parseQueryStr(postData));
            })
        } catch (error) {
            reject(error)
        }
    })
}

app.use(async (ctx) => {
    // 当请求时GET请求时，显示表单让用户填写
    if (ctx.url === '/' && ctx.method === 'GET') {
        ctx.body = `
            <form method="POST" action="/">
                <p>userName</p>
                <input name="userName"/> <br/>
                <p>passWord</p>
                <input name="passWord"/> <br/>
                <button type="submit">submit</button>
            </form>
        `;
        // 当请求时POST请求时
    } else if (ctx.url === '/' && ctx.method === 'POST') {
        ctx.body = await parseData(ctx);
    } else {
        // 其它请求显示404页面
        ctx.body = '<h1>404!</h1>';
    }
})

const route = async (url) => {
    let page = '';

        case '/':
            page = 'index.html';
            break;
        case '/index':
            page = 'index.html';
            break;
        case '/todo':
            page = 'todo.html';
            break;
        case '/404':
            page = '404.html';
            break;
        default:
            break;
    }

    // return html;

}

const renderHtml = page => {
    return new Promise((resolve, reject) => {
        // 假设静态html在page文件夹中
        let pageUrl = `./page/${page}`;
        // 原生fs模块读取正确路径的html，第二个参数为读取方式，binary为二进制
        fs.readFile(pageUrl, "binary", (err, data) => {
            if (err) {
                reject(err)
            } else {
                resolve(data);
            }
        })
    })
}
