# 2. GET请求

const Koa = require('koa')
const app = new Koa()

// 获得GET请求的方式有两种，一种是从request中获得，一种是一直从上下文中获得。
// 获得的格式也有两种：query和querystring。

app.use(async (ctx) => {
    // 得到请求地址
    let url = ctx.url;
    // 从request中获取GET请求
    let request = ctx.request;
    // 格式化好的参数对象
    let req_query = request.query;
    // 请求字符串
    let req_querystring = request.querystring;

    // 从上下文中直接获取
    let ctx_query = ctx.query;
    let ctx_querystring = ctx.querystring;

    ctx.body = {
        url,
        req_query,
        req_querystring,
        ctx_query,
        ctx_querystring
    }
})

app.listen(3000, () => {
    console.log('[demo] server is starting at port 3000');
});

// 输入http://localhost:3000/?user=lck&age=24就能看到参数等信息

# 3. POST请求

const Koa = require('koa')
const app = new Koa()

// 获取Post请求的步骤：
// 解析上下文ctx中的原生nodex.js对象req。
// 将POST表单数据解析成query string-字符串.(例如:user=jspang&age=18)
// 将字符串转换成JSON格式。

// ctx.request和ctx.req的区别
// request是Koa2中context经过封装的请求对象,req是context提供的node.js原生HTTP请求对象

// 通过处理原生req来处理参数
const parseData = ctx => {
    // 把字符串转化为json对象
    const parseQueryStr = queryStr => {
        let queryData = {};
        let queryStrList = queryStr.split('&');
        for (let [index,item] of queryStrList.entries()) {
            let obj = item.split('=');
            queryData[obj[0]] = decodeURIComponent(obj[1]);
        }
        return queryData
    }
    return new Promise((resolve, reject) => {
        try {
            let postData = '';
            ctx.req.on('data', (data) => {
                postData += data
            })
            ctx.req.addListener("end", () => {
                resolve(parseQueryStr(postData));
            })
        } catch (error) {
            reject(error)
        }
    })
}

app.use(async (ctx) => {
    // 当请求时GET请求时，显示表单让用户填写
    if (ctx.url === '/' && ctx.method === 'GET') {
        ctx.body = `
            <form method="POST" action="/">
                <p>userName</p>
                <input name="userName"/> <br/>
                <p>passWord</p>
                <input name="passWord"/> <br/>
                <button type="submit">submit</button>
            </form>
        `;
        // 当请求时POST请求时
    } else if (ctx.url === '/' && ctx.method === 'POST') {
        ctx.body = await parseData(ctx);
    } else {
        // 其它请求显示404页面
        ctx.body = '<h1>404!</h1>';
    }
})

app.listen(3000, () => {
    console.log('[demo] server is starting at port 3000');
});

# 4. Koa-bodyparser中间件

// npm i --save koa-bodyparser@3,4版本还需要安装各种库
// 中间件相当于引入一个插件，然后use调用即可
const Koa = require('koa')
const bodyParser = require('koa-bodyparser');

const app = new Koa()

app.use(bodyParser());

app.use(async (ctx) => {
    if (ctx.url === '/' && ctx.method === 'GET') {
        ctx.body = `
            <form method="POST" action="/">
                <p>userName</p>
                <input name="userName"/> <br/>
                <p>passWord</p>
                <input name="passWord"/> <br/>
                <button type="submit">submit</button>
            </form>
        `;
    } else if (ctx.url === '/' && ctx.method === 'POST') {
        // 直接可以用ctx.request.body进行获取POST请求参数，中间件自动给我们作了解析
        ctx.body = ctx.request.body;
    } else {
        ctx.body = '<h1>404!</h1>';
    }
})

app.listen(3000, () => {
    console.log('[demo] server is starting at port 3000');
});

# 5. Koa2原生路由

const Koa = require('koa')
const fs = require('fs')
const app = new Koa()

app.use(async (ctx) => {
    ctx.body = await route(ctx.request.url)
})

const route = async (url) => {
    let page = '';
    switch (url) {
        case '/':
            page = 'index.html';
            break;
        case '/index':
            page = 'index.html';
            break;
        case '/todo':
            page = 'todo.html';
            break;
        case '/404':
            page = '404.html';
            break;
        default:
            break;
    }
    // 找到对应的html静态页面
    // let html = await renderHtml(page);
    // return html;

    return page
}

const renderHtml = page => {
    return new Promise((resolve, reject) => {
        // 假设静态html在page文件夹中
        let pageUrl = `./page/${page}`;
        // 原生fs模块读取正确路径的html，第二个参数为读取方式，binary为二进制
        fs.readFile(pageUrl, "binary", (err, data) => {
            if (err) {
                reject(err)
            } else {
                resolve(data);
            }
        })
    })
}

app.listen(3000, () => {
    console.log('[demo] server is starting at port 3000');
});

# 6. Koa-router中间件

// npm i --save koa-router
const Koa = require('koa');
const Router = require('koa-router');
const app = new Koa();
// https://www.npmjs.com/package/koa-router
// 总路由
const router = new Router();

// 设置前缀层级路由
// const router = new Router({
//     prefix: '/lck'
// });

// 配置多路由
router
    .get('/', async (ctx, next) => {
        // 接受get路由参数
        ctx.body = ctx.query;
    })
    .get('/api', async (ctx, next) => {
        ctx.body = "api";
    });

// 配置子路由
let home = new Router();
home
    .get('/lck', async (ctx) => {
        ctx.body = "home lck";
    })
    .get('/todo', async (ctx) => {
        ctx.body = 'home ToDo';
    })

let page = new Router();
page
    .get('/lck', async (ctx) => {
        ctx.body = "page lck";
    })
    .get('/todo', async (ctx) => {
        ctx.body = 'page ToDo';
    })

// 装载所有子路由
router.use('/home', home.routes(), home.allowedMethods());
router.use('/page', page.routes(), page.allowedMethods());

// 加载中间件，routes()启动路由
// 官方文档的推荐用法,router.allowedMethods()用在了路由匹配router.routes()之后,
// 所以在当所有路由中间件最后调用.此时根据ctx.status设置response响应头
app
    .use(router.routes())
    .use(router.allowedMethods());

app.listen(3000, () => {
    console.log('[demo] server is starting at port 3000');
});

# 7. Koa2中使用cookie

const Koa = require('koa');
const app = new Koa();

// 设置Cookie
app.use(async (ctx) => {
    if (ctx.url === '/index') {
        // 写入cookie,第1，2个参数为value:key值，第3个为具体配置项
        ctx.cookies.set(
            'MyName', 'lck', {
                domain: '127.0.0.1', // 写cookie所在的域名
                path: '/index',       // 写cookie所在的路径
                maxAge: 1000 * 60 * 60 * 24,   // cookie有效时长
                expires: new Date('2018-12-31'), // cookie失效时间
                httpOnly: false,  // 是否只用于http请求中获取
                overwrite: false  // 是否允许重写
            }
        );
        ctx.body = 'cookie is ok';
    } else {
        // 读取cookie,在/index/xxx中可以读取cookie
        if (ctx.cookies.get('MyName')) {
            ctx.body = ctx.cookies.get('MyName');
        } else {
            ctx.body = 'Cookie is none';
        }
    }
});

app.listen(3000, () => {
    console.log('[demo] server is starting at port 3000');
})

# 8. ejs模版

// npm install --save koa-views
// npm install --save ejs
const Koa = require('koa')
const views = require('koa-views')
const path = require('path')
const app = new Koa()

// 加载模板引擎
app.use(views(path.join(__dirname, './view'), {
    extension: 'ejs'
}))

app.use(async (ctx) => {
    let title = 'hello koa2'
    await ctx.render('index', {
        title
    })
})

app.listen(3000, () => {
    console.log('[demo] server is starting at port 3000');
})

# 9. koa-static静态资源中间件

// npm install --save koa-static
const Koa = require('koa')
const path = require('path')
const static = require('koa-static')
const app = new Koa()

// 读取static目录中的静态资源文件，比如http://localhost:3000/house.jpg
app.use(static(
    path.join(__dirname, './static')
))

app.use(async (ctx) => {
    ctx.body = 'hello world'
})

app.listen(3000, () => {
    console.log('[demo] server is starting at port 3000')
})
