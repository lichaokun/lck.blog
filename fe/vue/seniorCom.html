# 1-1 notification.vue

<template>
    <transition name="fade" @after-leave="afterLeave" @after-enter="afterEnter">
        <div
            class="notification"
            :style="style"
            v-show="visible"
            @mouseenter="clearTimer"
            @mouseleave="createTimer"
        >
            <span class="content">{{content}}</span>
            <a class="btn" @click="handleClose">{{btn}}</a>
        </div>
    </transition>
</template>

<script>
export default {
    // 组件名
    name: 'Notification',
    props: {
        // 通知栏内容
        content: {
            type: String,
            required: true
        },
        // 按钮文字
        btn: {
            type: String,
            default: '关闭'
        }
    },
    data () {
        return {
            // 显示/隐藏
            visible: true
        }
    },
    computed: {
        // 默认返回一个空样式，通过func-notification继承来的样式覆盖掉当前组件的样式
        style () {
            return {}
        }
    },
    methods: {
        // 点击关闭
        handleClose (e) {
            e.preventDefault()
            this.$emit('close')
        },
        // 动画完成后
        afterLeave () {
            this.$emit('closed')
        },
        // 通过func-notification继承过来的方法都要在这个组件中预先设定好
        afterEnter () {},
        clearTimer () {},
        createTimer () {}
    }
}
</script>

# 1-2 func-notification.js

import Notification from './notification.vue'

// 通过extends来继承拓展组件
export default {
    extends: Notification,
    computed: {
        // 组件样式
        style () {
            return {
                position: 'fixed',
                right: '20px',
                bottom: `${this.verticalOffset}px`
            }
        }
    },
    mounted () {
        this.createTimer()
    },
    methods: {
        // 创建定时器
        createTimer () {
            console.log(this.autoClose)
            if (this.autoClose) {
                this.timer = setTimeout(() => {
                    this.visible = false
                }, this.autoClose)
            }
        },
        // 清除定时器
        clearTimer () {
            if (this.timer) {
                clearTimeout(this.timer)
            }
        },
        afterEnter () {
            this.height = this.$el.offsetHeight
        }
    },
    beforeDestory () {
        this.clearTimer()
    },
    data () {
        return {
            verticalOffset: 0,
            autoClose: 3000,
            height: 0,
            visible: false
        }
    }
}

# 1-3 function.js

import Vue from 'vue'
import Component from './func-notification'

// 继承组件
const NotificationConstructor = Vue.extend(Component)

// 实例数组
const instances = []
// 组件ID
let seed = 1

// 删除实例
const removeInstance = (instance) => {
    if (!instance) return
    const len = instances.length
    const index = instances.findIndex(inst => instance.id === inst.id)

    instances.splice(index, 1)

    // 调整其他实例的高度
    if (len <= 1) return
    const removeHeight = instance.vm.height
    for (let i = index; i < len - 1; i++) {
        instances[i].verticalOffset = parseInt(instances[i].verticalOffset) - removeHeight - 16
    }
}

// 生成实例
const notify = (options) => {
    // 如果是服务端，就返回，因为无法创立DOM
    if (Vue.prototype.$isServer) return

    // 生成实例时候传递的参数
    const {
        autoClose,
        ...rest
    } = options
    const instance = new NotificationConstructor({
        // 把autoClose当作data传入，其余的参数当作options传入
        propsData: {
            ...rest
        },
        data: {
            autoClose: autoClose == undefined ? 3000 : autoClose
        }
    })

    // 生成不重复的ID实例
    const id = `notification_${seed++}`
    instance.id = id
    // 插入DOM接点
    instance.vm = instance.$mount()
    document.body.appendChild(instance.vm.$el)
    // 显示实例
    instance.vm.visible = true

    // 计算高度
    let verticalOffset = 0
    instances.forEach(item => {
        verticalOffset += item.$el.offsetHeight + 16
    })
    verticalOffset += 16
    instance.verticalOffset = verticalOffset
    // 记录实例
    instances.push(instance)
    // 动画完成后
    instance.vm.$on('closed', () => {
        // 删除实例
        removeInstance(instance)
        // 删除DOM接点
        document.body.removeChild(instance.vm.$el)
        // destroy组件
        instance.vm.$destroy()
    })
    // 点击关闭后
    instance.vm.$on('close', () => {
        // 隐藏实例
        instance.vm.visible = false
    })
    return instance.vm
}

export default notify

# 2. tab组件, 使用react的render写法来编写灵活度更高的组件

# 2-1 tab-container.vue

<script>
export default {
  props: {
    panes: {
      type: Array,
      required: true
    }
  },
  render (h) {
    const contents = this.panes.map(pane => {
      return pane.active ? pane.$slots.default : null
    })
    return (
      <div class="tab-content">
        {contents}
      </div>
    )
  }
}
</script>

# 2-2 tab.vue

<script>
export default {
  name: 'Tab',
  props: {
    index: {
      required: true,
      type: [Number, String]
    },
    label: {
      type: String,
      default: 'tab'
    }
  },
  mounted () {
    this.$parent.panes.push(this)
  },
  computed: {
    active () {
      return this.$parent.value === this.index
    }
  },
  methods: {
    handleClick () {
      this.$parent.onChange(this.index)
    }
  },
  render (h) {
    const tab = this.$slots.label || <span>{this.label}</span>
    const classNames = {
      tab: true,
      active: this.active
    }
    return (
      <li class={classNames} on-click={this.handleClick}>
        {tab}
      </li>
    )
  }
}
</script>

# 2-3 tabs.vue

<script>
import TabContainer from './tab-container.vue'

export default {
  name: 'Tabs',
  components: {
    TabContainer
  },
  props: {
    value: {
      type: [String, Number],
      required: true
    }
  },
  data () {
    return {
      panes: []
    }
  },
  render (h) {
    return (
      <div class='tabs'>
        <ul class='tabs-header'>
          {this.$slots.default}
        </ul>
        <tab-container panes={this.panes}></tab-container>
      </div>
    )
  },
  methods: {
    onChange (index) {
      this.$emit('change', index)
    }
  }
}
</script>
