# 1. Array.includes 与条件判断

// 一般我们判断或用 ||
function test(fruit) {
  if (fruit == "apple" || fruit == "strawberry") {
    console.log("red");
  }
}

// 判断多个条件
function test(fruit) {
  const redFruits = ["apple", "strawberry", "cherry", "cranberries"];

  if (redFruits.includes(fruit)) {
    console.log("red");
  }
}

test('apple', 'cherry')  
// red

# 2. Set去重

// 数组去重
const arr = [3, 5, 2, 2, 5, 5];
const unique = [...new Set(arr)]; 
// [3,5,2]

// Array.from 方法可以将 Set 结构转为数组
function unique(array) {
  return Array.from(new Set(array));
}

unique([1, 1, 2, 3]); 
// [1, 2, 3]

// 字符去重
const str = [...new Set("ababbc")].join("");
console.log(str); 
// 'abc'


// 并集 交集 差集
let a = new Set([1, 2, 3]);
let b = new Set([4, 3, 2]);

// 并集
let union = new Set([...a, ...b]);
// Set {1, 2, 3, 4}

// 交集
let intersect = new Set([...a].filter(x => b.has(x)));
// set {2, 3}

// 差集
let difference = new Set([...a].filter(x => !b.has(x)));
// Set {1}

# 3. Map 与字典类型数据

js 实现字典数据是基于 Object 对象。对象的键只能是字符串。ES6 提供了 Map 数据结构。它类似于 Object 对象，也是键值对的集合，但是“键”的范围不限于字符串，各种类型的值，字符串、数值、布尔值、数组、对象等等都可以当作键。

// 普通map
const map = new Map()
  .set(-1, {text:'小于',color:'yellow'})
  .set(0, {text:'等于',color:'black'})
  .set(1, {text:'大于',color:'green'})
  .set(null,{text:'没有物品',color:'red'})

const state = map.get(null)
// {text:'没有物品',color:'red'}

// Map 的遍历顺序就是插入顺序
const map = new Map([["F", "no"], ["T", "yes"]]);

for (let key of map.keys()) {
  console.log(key);
}
// "F"
// "T"

for (let value of map.values()) {
  console.log(value);
}
// "no"
// "yes"

# 4. 函数式的方式处理数据

let arr = [
    {
        name: "Stark",
        age: 21
    },
    {
        name: "Jarvis",
        age: 20
    },
    {
        name: "Pepper",
        age: 16
    }
];

//  希望得到的数据结构
[
    {
        person: {
            name: "Stark",
            age: 21
        },
        friends: []
    },
    {
        person: {
            name: "Jarvis",
            age: 20
        },
        friends: []
    }
]

// 普通写法
let result = [];
arr.forEach(function(person){
    if(person.age > 16){
        let newItem = {
            person: person,
            friends: []
        };
        result.push(newItem);
    }
})

// 函数写法
let result = arr
  .filter(person => person.age > 16)
  .map(person => ({
    person: person,
    friends: []
  }));

// 数组求和
let foo = [1, 2, 3, 4, 5];

// 普通写法
function sum(arr) {
  let x = 0;
  for (let i = 0; i < arr.length; i++) {
    x += arr[i];
  }
  return x;
}
sum(foo); 
// 15

// 函数写法
foo.reduce((a, b) => a + b, 0); 
// 15

# 5. compose 与函数组合

compose 就是结合函数，生成一个新的函数

// 普通compose,在 compose 的定义中， g 将先于 f 执行，因此就创建了一个从右到左的数据流。可读性远远高于嵌套一大堆的函数调用.
const compose = function(f, g) {
  return function(x) {
    return f(g(x));
  };
};

// es6版本，只能处理两个函数参数
const compose = (f, g) => x => f(g(x));

// 利用compose生成一个新的函数，先反转然后取第一个元素
const head = arr => arr[0];
const reverse = arr => [].concat(arr).reverse();

const last = compose(head, reverse);
last(["jumpkick", "roundhouse", "uppercut"]);
// "uppercut"

// redux源码compose， 可以随意组合无数个函数
function compose(...funcs){
    if (funcs.length === 0) return arg => arg;

    if (funcs.length === 1 ) return funcs[0];

    return funcs.reduce((a,b) => (...args) => a(b(...args)))
}

// 组合2个和以上的函数
const head = arr => arr[0];
const reverse = arr => [].concat(arr).reverse();
const toUpperCase = str => str.toUpperCase();

const last = compose(head, reverse);
const lastAndUpper = compose(toUpperCase, head, reverse);

console.log(last(["jumpkick", "roundhouse", "uppercut"]))
// "uppercut"
console.log(lastAndUpper(["jumpkick", "roundhouse", "uppercut"]))
// "UPPERCUT"
