# 第1题

function foo() {
    console.log(this.a);
}
function doFoo(fn) {
    fn(); 
}
var obj = {
    a: 2,
    foo: foo
};
var a = "I’m an a";

// 函数调用this指向window
doFoo(obj.foo); // I’m an a

# 第2题

function foo(something) {
    console.log(this.a, something);
    return this.a + something;
}

function bind(fn, obj) {
    return function() {
        return fn.apply(obj, arguments);
    };
}
var obj = {
    a: 2
};
var bar = bind(foo, obj);
var b = bar(3);

// bar此时为bind函数内部返回的函数，且传了foo函数和obj为参数, 即 return foo.apply({a: 2}}, arguments), 当调用bar(3)时, 3即为arguments, 所以等于2+3 = 5
console.log(b); // 5

# 第3题

function foo(something) {
    this.a = something;
}
var obj1 = {
    foo: foo
};
var obj2 = {};

obj1.foo(2);
console.log(obj1.a); // 2

obj1.foo.call(obj2, 3);
console.log(obj2.a); // 3

var bar = new obj1.foo(4); // new运算符并没有修改obj1.a的值
console.log(obj1.a); // 2
console.log(bar.a); // 4

# 第4题

function foo(something) {
    this.a = something;
}
var obj1 = {};
var bar = foo.bind(obj1);
bar(2);
console.log(obj1.a); // 2

var baz = new bar(3); // new运算符并没有修改obj1.a的值， 3等于后续的传参
console.log(obj1.a); // 2
console.log(baz.a); // 3

# 第5题

function foo(p1,p2) {
    this.val = p1 + p2;
}
var bar = foo.bind(null, "p1"); // bind，第一个传null得话不改变this指向，而且可以在后续的调用中去传入参数, 相当于是'p1'已经传入了参数
var baz = new bar("p2"); // 第二个参数追加

console.log(baz.val); // 'p1p2'
